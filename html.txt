<script>
let squads = JSON.parse(localStorage.getItem('pak_squads')) || {};
let m = { r:0, w:0, b:0, p1:{n:"",r:0,b:0}, p2:{n:"",r:0,b:0}, bwl:{n:"",r:0,w:0,b:0}, stk:1, hist:[], ovB:[] };

function saveT() {
  let id = document.getElementById('slot-idx').value;
  let n = document.getElementById('t-name').value;
  let p = document.getElementById('t-p').value.split(',').map(x=>x.trim());
  if(!n) return alert("Team name likhain!");
  squads[id] = { n, p };
  localStorage.setItem('pak_squads', JSON.stringify(squads));
  updateDrops(); alert("Team Saved!");
}

function updateDrops() {
  let h = Object.keys(squads).map(id => `<option value="${id}">${squads[id].n}</option>`).join('');
  document.getElementById('sel-t1').innerHTML = h; document.getElementById('sel-t2').innerHTML = h;
}

function openToss() {
  let t1 = squads[document.getElementById('sel-t1').value].n;
  let t2 = squads[document.getElementById('sel-t2').value].n;
  showM("Who won the Toss?", [t1, t2], winner => {
    showM(winner + " decided to?", ["BAT FIRST", "BOWL FIRST"], dec => {
      m.batTeam = (dec==="BAT FIRST" && winner===t1) || (dec==="BOWL FIRST" && winner===t2) ? squads[document.getElementById('sel-t1').value] : squads[document.getElementById('sel-t2').value];
      m.bowlTeam = m.batTeam.n === t1 ? squads[document.getElementById('sel-t2').value] : squads[document.getElementById('sel-t1').value];
      startMatchFlow();
    });
  });
}

function startMatchFlow() {
  document.getElementById('setup-view').style.display='none';
  document.getElementById('score-view').style.display='block';
  document.getElementById('tv-bar').style.display='block';
  document.getElementById('ui-team-name').innerText = m.batTeam.n;
  showM("Select Opening Batsman 1", m.batTeam.p, n1 => { m.p1.n=n1;
    showM("Select Opening Batsman 2", m.batTeam.p.filter(x=>x!==n1), n2 => { m.p2.n=n2;
      showM("Select Opening Bowler", m.bowlTeam.p, b => { m.bwl.n=b; updateUI(); });
    });
  });
}

function askExtra(type) {
  showM(type + " + How many runs?", ["0","1","2","3","4","6"], r => {
    saveH(); let runs = parseInt(r); m.r += (runs+1); m.bwl.r += (runs+1);
    m.ovB.push({l:type+(runs>0?'+'+runs:''), c:'orange'});
    updateUI();
  });
}

function askWicket() {
  showM("Wicket Type?", ["Bowled", "Caught", "Run Out", "Stumped", "LBW"], type => {
    saveH(); m.w++; m.b++; m.bwl.b++; if(type!=="Run Out") m.bwl.w++;
    m.ovB.push({l:'W', c:'#dc2626'});
    showM("Next Batsman?", m.batTeam.p, n => {
      if(m.stk===1) m.p1={n:n,r:0,b:0}; else m.p2={n:n,r:0,b:0};
      checkOver(); updateUI();
    });
  });
}

function ball(r) {
  saveH(); m.r+=r; m.b++; m.bwl.r+=r; m.bwl.b++;
  let cur = m.stk===1?m.p1:m.p2; cur.r+=r; cur.b++;
  m.ovB.push({l:r, c: r===6?'#7c3aed':(r===4?'#2563eb':'#444')});
  if(r%2!==0) m.stk=m.stk===1?2:1;
  checkOver(); updateUI();
}

function checkOver() {
  if(m.b%6===0 && m.b>0) {
    m.stk=m.stk===1?2:1;
    setTimeout(() => { m.ovB=[]; showM("Select New Bowler", m.bowlTeam.p, b => { m.bwl={n:b,r:0,w:0,b:0}; updateUI(); }); }, 800);
  }
}

function showM(title, opts, cb) {
  let ov = document.getElementById('modal-overlay'); ov.style.display='block';
  document.getElementById('modal-title').innerText = title;
  document.getElementById('modal-content').innerHTML = opts.map(o => `<button class="modal-btn" onclick="selM('${o}')">${o}</button>`).join('');
  window.mCB = cb;
}
function selM(v) { document.getElementById('modal-overlay').style.display='none'; window.mCB(v); }
function saveH() { m.hist.push(JSON.stringify(m)); }
function undo() { if(m.hist.length>0) { m = JSON.parse(m.hist.pop()); updateUI(); } }

function updateUI() {
  document.getElementById('ui-runs').innerText = m.r; document.getElementById('ui-wkts').innerText = m.w;
  document.getElementById('ui-ovs').innerText = Math.floor(m.b/6) + "." + (m.b%6);
  document.getElementById('ui-bat1').innerText = m.p1.n + (m.stk===1?'* ':' ') + m.p1.r + "(" + m.p1.b + ")";
  document.getElementById('ui-bat2').innerText = m.p2.n + (m.stk===2?'* ':' ') + m.p2.r + "(" + m.p2.b + ")";
  document.getElementById('ui-bwl-name').innerText = m.bwl.n;
  document.getElementById('ui-bwl-stat').innerText = `${m.bwl.w}-${m.bwl.r} (${Math.floor(m.bwl.b/6)}.${m.bwl.b%6})`;
  document.getElementById('balls-list').innerHTML = m.ovB.map(b => `<div style="width:25px; height:25px; border-radius:50%; background:${b.c}; color:white; display:flex; align-items:center; justify-content:center; font-size:10px; font-weight:bold;">${b.l}</div>`).join('');
}
updateDrops();
</script>
